## 문제

```javascript
/*
주어진 행렬(arr)에서 가장 낮은 수를 찾아서 그 수를 제외한 행렬을 리턴하라
*/

// 예시
arr = [1,2,3,4] 
answer = [2,3,4]
```

## 정답

```javascript
function solution(arr) {
  let answer = [];
  let min = arr.reduce((a,b) => a > b ? b : a)
  arr.forEach((a,i) => {
      if(a !== min) answer.push(a)
  })
  if (answer.length === 0) return [-1]
  return answer;
}
```

## 문제 

```javascript
/*
체육복 빌려주기
옆사람에게만 체육복 빌려줄 수 있음.
주어지는 인자는 n,lost,reserve
n은 총 사람수
lost는 체육복 없는 사람
reserve는 체육복이 2개 있는 사람.
나머지는 체육복 1개만 있는 사람들.
*/

//예시
n = 5
lost = [2,4]
reserve = [1,3,5]

answer = 5
```

## 정답

```javascript
function solution(n, lost, reserve) {
  let lostReserved = new Set()
  let reserveSet = new Set(reserve)
  lost.forEach(l => {
      lostReserved.add(l-1)
      lostReserved.add(l+1)
  })
  let noOfReserved = 0
  reserve.forEach(ls => {
      if(lostReserved.has(ls)) noOfReserved++
  })
  if(lost.length < noOfReserved) noOfReserved = lost.length
  let answer = n - lost.length + noOfReserved
  return answer
}
```

## 문제

```javascript
/*
  최대 공약수의 합 구하기
  n = 12
  [1,2,3,4,6,12] => 38
*/

function solution(n) {
  if(n <= 1) return n
  let last
  let answer = 0
  for(let i=1;i<=n;i++) {
      if(n%i===0) {
          let a = n/i
          if(last === a) break
          last = i
          if(i===a) return answer + i
          answer += i + a
      }
  }

  return answer
}
```

## 문제


```javascript
/*
  두 수(n,m)의 최대공약수와 최소공배수 구하기

  n = 1
  m = 5
  answer = [1, 10]
*/
```

## 정답

```javascript
function solution(n, m) {

  let a = n > m ? n : m
  let b = n < m ? n : m

  let GCD
  let set = new Set()
  for(let i=1;i<=b;i++) {
      if(b%i == 0) {
          set.add(i)
      }
  }
  set.forEach(s => {
      if(a%s ==0) GCD = s
  })

  let LCM
  if(a%b == 0) LCM = a
  else LCM = a*b/GCD

  return [GCD, LCM]
}
```
